  <script>

    // Data store (IndexedDB wrapper for persistence)

    let db;

    const openReq = indexedDB.open('sprucheDB',1);

    openReq.onupgradeneeded = e => {

      db = e.target.result;

      const store = db.createObjectStore('verses',{keyPath:'id'});

      store.createIndex('text','iast',{unique:false});

    };

    openReq.onsuccess = e => {db = e.target.result; loadVerse(1);} ;


    // Elements

    const iastEl = document.getElementById('iast');

    const devEl  = document.getElementById('dev');


    // Real‑time transliteration via Aksharamukha

    iastEl.addEventListener('input', ()=>{

      devEl.value = Aksharamukha.convert('IAST','Devanagari',iastEl.value.trim());

    });


    // Save verse

    document.getElementById('verseForm').addEventListener('submit',e=>{

      e.preventDefault();

      const verse = {

        id: Number(document.getElementById('verseId').value),

        iast:iastEl.value.trim(),

        dev:devEl.value.trim(),

        de:document.getElementById('de').value.trim(),

        ja:document.getElementById('ja').value.trim(),

        notes:document.getElementById('notes').value.trim()

      };

      const tx = db.transaction('verses','readwrite');

      tx.objectStore('verses').put(verse);

      alert('Saved verse '+verse.id);

    });


    // Navigation helpers

    document.getElementById('prev').onclick = ()=>jump(-1);

    document.getElementById('next').onclick = ()=>jump(1);

    document.getElementById('rand').onclick = ()=>{

      loadVerse(Math.floor(Math.random()*7613)+1);

    };

    function jump(delta){

      const id = Number(document.getElementById('verseId').value)||1;

      loadVerse(Math.max(1,Math.min(7613,id+delta)));

    }


    function loadVerse(id){

      const tx = db.transaction('verses');

      const req = tx.objectStore('verses').get(id);

      req.onsuccess = () => {

        const v=req.result||{id, iast:'',dev:'',de:'',ja:'',notes:''};

        document.getElementById('verseId').value=v.id;

        iastEl.value=v.iast; devEl.value=v.dev;

        document.getElementById('de').value=v.de;

        document.getElementById('ja').value=v.ja;

        document.getElementById('notes').value=v.notes;

      };

    }


    // Search

    document.getElementById('searchBtn').onclick = ()=>{

      const term=document.getElementById('searchInput').value.toLowerCase();

      const list=[];

      const tx=db.transaction('verses');

      tx.objectStore('verses').openCursor().onsuccess=e=>{

        const c=e.target.result; if(!c)return render(list);

        const v=c.value;

        const all=(v.iast+v.dev+v.de+v.ja+v.notes).toLowerCase();

        if(all.includes(term)) list.push(v);

        c.continue();

      };

    };

    function render(arr){

      const cont=document.getElementById('verseContainer');

      cont.innerHTML= arr.length? arr.map(v=>`<div class='verseCard'><strong>${v.id}</strong> ${v.dev||v.iast}</div>`).join('') : '<p>No match</p>';

    }
