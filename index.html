    <script>

        // Application state

        let verses = {};

        let currentVerseId = 1;

        const totalExpectedVerses = 7600;


        // Initialize application

        document.addEventListener('DOMContentLoaded', function() {

            loadFromMemory();

            updateStats();

            updateDisplay();

            

            // Event listeners

            document.getElementById('verseForm').addEventListener('submit', saveVerse);

            document.getElementById('clearForm').addEventListener('click', clearForm);

            document.getElementById('loadVerse').addEventListener('click', loadCurrentVerse);

            document.getElementById('gotoBtn').addEventListener('click', gotoVerse);

            document.getElementById('searchBtn').addEventListener('click', searchVerses);

            document.getElementById('prevVerse').addEventListener('click', () => navigateVerse(-1));

            document.getElementById('nextVerse').addEventListener('click', () => navigateVerse(1));

            document.getElementById('randomVerse').addEventListener('click', randomVerse);

            document.getElementById('exportJSON').addEventListener('click', exportJSON);

            document.getElementById('exportHTML').addEventListener('click', exportHTML);

            document.getElementById('importJSON').addEventListener('click', () => document.getElementById('importFile').click());

            document.getElementById('importFile').addEventListener('change', importJSON);

            

            // Auto-save functionality

            const inputs = ['devanagariText', 'romanText', 'germanTranslation', 'japaneseTranslation', 'notes'];

            inputs.forEach(id => {

                document.getElementById(id).addEventListener('input', debounce(autoSave, 2000));

            });

        });


        function saveVerse(e) {

            e.preventDefault();

            

            const verseId = parseInt(document.getElementById('verseNumber').value);

            const verseData = {

                id: verseId,

                devanagari: document.getElementById('devanagariText').value.trim(),

                roman: document.getElementById('romanText').value.trim(),

                german: document.getElementById('germanTranslation').value.trim(),

                japanese: document.getElementById('japaneseTranslation').value.trim(),

                notes: document.getElementById('notes').value.trim(),

                lastModified: new Date().toISOString()

            };

            

            verses[verseId] = verseData;

            currentVerseId = verseId;

            

            saveToMemory();

            updateStats();

            updateDisplay();

            

            // Show success feedback

            showNotification(Verse ${verseId} saved successfully!, 'success');

            

            // Auto-advance to next verse

            if (verseId < totalExpectedVerses) {

                setTimeout(() => {

                    document.getElementById('verseNumber').value = verseId + 1;

                    currentVerseId = verseId + 1;

                    clearForm(false);

                }, 1000);

            }

        }


        function clearForm(clearNumber = true) {

            if (clearNumber) {

                document.getElementById('verseNumber').value = '';

            }

            document.getElementById('devanagariText').value = '';

            document.getElementById('romanText').value = '';

            document.getElementById('germanTranslation').value = '';

            document.getElementById('japaneseTranslation').value = '';

            document.getElementById('notes').value = '';

        }


        function loadCurrentVerse() {

            const verseId = parseInt(document.getElementById('verseNumber').value);

            if (verses[verseId]) {

                const verse = verses[verseId];

                document.getElementById('devanagariText').value = verse.devanagari || '';

                document.getElementById('romanText').value = verse.roman || '';

                document.getElementById('germanTranslation').value = verse.german || '';

                document.getElementById('japaneseTranslation').value = verse.japanese || '';

                document.getElementById('notes').value = verse.notes || '';

                currentVerseId = verseId;

                showNotification(Verse ${verseId} loaded, 'info');

            } else {

                showNotification(Verse ${verseId} not found, 'warning');

            }

        }


        function gotoVerse() {

            const verseId = parseInt(document.getElementById('gotoVerse').value);

            if (verseId >= 1 && verseId <= totalExpectedVerses) {

                document.getElementById('verseNumber').value = verseId;

                currentVerseId = verseId;

                loadCurrentVerse();

            }

        }


        function navigateVerse(direction) {

            const newId = Math.max(1, Math.min(totalExpectedVerses, currentVerseId + direction));

            document.getElementById('verseNumber').value = newId;

            currentVerseId = newId;

            loadCurrentVerse();

        }


        function randomVerse() {

            const verseId = Math.floor(Math.random() * totalExpectedVerses) + 1;

            document.getElementById('verseNumber').value = verseId;

            currentVerseId = verseId;

        function searchVerses() {

            const results = Object.values(verses).filter(verse =>

    function transliterateIAST(input) {

        const sortedKeys = Object.keys(iastToDevanagariMap).sort((a, b) => b.length - a.length);

        let output = '';

        let i = 0;

        while (i < input.length) {

            let matched = false;

            for (let key of sortedKeys) {

                if (input.substring(i, i + key.length) === key) {

                    output += iastToDevanagariMap[key];

                    i += key.length;

                    matched = true;

                    break;

                }

            }

            if (!matched) {

                output += input[i];

                i++;

            }

        }

        return output;

    }
